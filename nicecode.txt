// ========== LEDC PWM INITIALIZATION ==========
void initLEDC() {
    ledcSetup(CH_LEFT_FORWARD, LEDC_BASE_FREQ, LEDC_TIMER_13_BIT);
    ledcSetup(CH_LEFT_BACKWARD, LEDC_BASE_FREQ, LEDC_TIMER_13_BIT);
    ledcSetup(CH_RIGHT_FORWARD, LEDC_BASE_FREQ, LEDC_TIMER_13_BIT);
    ledcSetup(CH_RIGHT_BACKWARD, LEDC_BASE_FREQ, LEDC_TIMER_13_BIT);
    
    ledcAttachPin(leftForwardPin, CH_LEFT_FORWARD);
    ledcAttachPin(leftBackwardPin, CH_LEFT_BACKWARD);
    ledcAttachPin(rightForwardPin, CH_RIGHT_FORWARD);
    ledcAttachPin(rightBackwardPin, CH_RIGHT_BACKWARD);
    
    ledcWrite(CH_LEFT_FORWARD, 0);
    ledcWrite(CH_LEFT_BACKWARD, 0);
    ledcWrite(CH_RIGHT_FORWARD, 0);
    ledcWrite(CH_RIGHT_BACKWARD, 0);
}

// ========== MOTOR CONTROL FUNCTIONS WITH PWM ==========
void stopMotors() {
    ledcWrite(CH_LEFT_FORWARD, 0);
    ledcWrite(CH_LEFT_BACKWARD, 0);
    ledcWrite(CH_RIGHT_FORWARD, 0);
    ledcWrite(CH_RIGHT_BACKWARD, 0);
    currentCommand = "stop";
    Serial.println("Command: STOP");
}

void moveForward(int speed) {
    speed = constrain(speed, 0, 255);
    ledcWrite(CH_LEFT_FORWARD, speed);
    ledcWrite(CH_LEFT_BACKWARD, 0);
    ledcWrite(CH_RIGHT_FORWARD, speed);
    ledcWrite(CH_RIGHT_BACKWARD, 0);
    currentCommand = "forward";
    Serial.printf("Command: FORWARD at speed %d\n", speed);
}

void moveBackward(int speed) {
    speed = constrain(speed, 0, 255);
    ledcWrite(CH_LEFT_FORWARD, 0);
    ledcWrite(CH_LEFT_BACKWARD, speed);
    ledcWrite(CH_RIGHT_FORWARD, 0);
    ledcWrite(CH_RIGHT_BACKWARD, speed);
    currentCommand = "backward";
    Serial.printf("Command: BACKWARD at speed %d\n", speed);
}

void moveLeft(int speed) {
    speed = constrain(speed, 0, 255);
    ledcWrite(CH_LEFT_FORWARD, 0);
    ledcWrite(CH_LEFT_BACKWARD, speed);
    ledcWrite(CH_RIGHT_FORWARD, speed);
    ledcWrite(CH_RIGHT_BACKWARD, 0);
    currentCommand = "left";
    Serial.printf("Command: LEFT at speed %d\n", speed);
}

void moveRight(int speed) {
    speed = constrain(speed, 0, 255);
    ledcWrite(CH_LEFT_FORWARD, speed);
    ledcWrite(CH_LEFT_BACKWARD, 0);
    ledcWrite(CH_RIGHT_FORWARD, 0);
    ledcWrite(CH_RIGHT_BACKWARD, speed);
    currentCommand = "right";
    Serial.printf("Command: RIGHT at speed %d\n", speed);
}

// ========== WEB SERVER HANDLERS ==========
void handleRoot() {
    server.send(200, "text/html", index_html);
}

void handleSpeed() {
    String speedStr = server.pathArg(0);
    int speed = speedStr.toInt();
    speed = constrain(speed, 0, 255);
    
    if (currentCommand != "stop") {
        if (currentCommand == "forward") moveForward(speed);
        else if (currentCommand == "backward") moveBackward(speed);
        else if (currentCommand == "left") moveLeft(speed);
        else if (currentCommand == "right") moveRight(speed);
    }
    
    server.send(200, "text/plain", String(speed));
}

void handleCommand() {
    String command = server.pathArg(0);
    String speedStr = server.pathArg(1);
    int speed = speedStr.length() > 0 ? speedStr.toInt() : DEFAULT_SPEED;
    speed = constrain(speed, 0, MAX_SPEED);
    
    if (command == "forward") {
        moveForward(speed);
        server.send(200, "text/plain", "FORWARD");
    } 
    else if (command == "backward") {
        moveBackward(speed);
        server.send(200, "text/plain", "BACKWARD");
    }
    else if (command == "left") {
        moveLeft(speed);
        server.send(200, "text/plain", "LEFT");
    }
    else if (command == "right") {
        moveRight(speed);
        server.send(200, "text/plain", "RIGHT");
    }
    else if (command == "stop") {
        stopMotors();
        server.send(200, "text/plain", "STOP");
    }
    else {
        server.send(400, "text/plain", "Unknown command");
    }
}

// ========== SETUP ==========
void setup() {
    Serial.begin(115200);
    delay(100);
    
    initLEDC();
    Serial.println("✅ LEDC PWM initialized for touch control");
    
    Serial.println();
    Serial.println("=== XIAO TOUCH ROBOT CONTROLLER ===");
    Serial.print("Connecting to Wi-Fi: ");
    Serial.println(ssid);
    
    WiFi.begin(ssid, password);
    
    int attempts = 0;
    while (WiFi.status() != WL_CONNECTED && attempts < 20) {
        delay(500);
        Serial.print(".");
        attempts++;
    }
    
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println();
        Serial.println("✅ WiFi Connected!");
        Serial.print("📡 IP Address: ");
        Serial.println(WiFi.localIP());
        Serial.println("🌐 Open this IP on your TOUCH device");
        Serial.println("🎮 Drag the joystick or tap buttons to control");
    } else {
        Serial.println();
        Serial.println("❌ WiFi connection failed!");
    }
    
    server.on("/", handleRoot);
    server.onNotFound([]() {
        String uri = server.uri();
        if (uri.startsWith("/speed/")) {
            handleSpeed();
        } else if (uri.startsWith("/cmd/")) {
            handleCommand();
        } else {
            server.send(404, "text/plain", "Not Found");
        }
    });
    
    server.begin();
    Serial.println("✅ Web server ready for touch input");
    Serial.println("==================================");
}

void loop() {
    server.handleClient();
    delay(10);
}
